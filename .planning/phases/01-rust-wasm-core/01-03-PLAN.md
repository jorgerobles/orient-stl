      export async function initWasm(): Promise<void> {
        await init();
        wasmReady = true;
        console.log('WASM initialized');
      }

      export async function loadSTLBytes(file: File): Promise<Uint8Array> {
        if (file.size === 0) throw new Error('Empty file');
        const buffer = await file.arrayBuffer();
        return new Uint8Array(buffer);
      }

      export async function processSTL(bytes: Uint8Array, config: OrientConfig): Promise<Candidate[]> {
        if (!wasmReady) throw new Error('WASM not initialized');

        return result;

    ```typescript
    import { initWasm, loadSTLBytes, processSTL } from './loadSTL';
    import type { Candidate, OrientConfig } from './types';
    import { defaultConfig } from './types';

    let config: OrientConfig = defaultConfig();
    let candidates: Candidate[] = [];

    // --- DOM refs ---
    const statusEl = document.getElementById('status')!;
    const fileInput = document.getElementById('file-input') as HTMLInputElement;
    const dropZone = document.getElementById('drop-zone')!;
    const resultsEl = document.getElementById('results')!;

    // --- WASM init ---
    async function boot(): Promise<void> {
      statusEl.textContent = 'Initializing WASM...';
      try {
        await initWasm();
        statusEl.textContent = 'Ready — load an STL file';
      } catch (err) {
        statusEl.textContent = `WASM init failed: ${err}`;
        console.error(err);
      }
    }

    // --- File processing ---
    async function handleFile(file: File): Promise<void> {
      if (!file.name.toLowerCase().endsWith('.stl')) {
        statusEl.textContent = 'Please select a .stl file';
        return;
      }
      statusEl.textContent = `Loading ${file.name}...`;
      try {
        const bytes = await loadSTLBytes(file);
        statusEl.textContent = `Processing ${file.name} (${bytes.length} bytes)...`;
        candidates = await processSTL(bytes, config);
        displayResults(candidates);
      } catch (err) {
        statusEl.textContent = `Error: ${err}`;
        console.error(err);
      }
    }

    // --- File picker ---
    fileInput.addEventListener('change', () => {
      const files = fileInput.files;
      if (files && files.length > 0) {
        handleFile(files[0]);
      }
    });

    // --- Drag-drop ---
    dropZone.addEventListener('dragover', (e) => {
      e.preventDefault();
      dropZone.classList.add('drag-over');
    });
    dropZone.addEventListener('dragleave', () => {
      dropZone.classList.remove('drag-over');
    });
    dropZone.addEventListener('drop', (e) => {
      e.preventDefault();
      dropZone.classList.remove('drag-over');
      const files = e.dataTransfer?.files;
      if (files && files.length > 0) {
        handleFile(files[0]);
      }
    });

    // --- Display results ---
    function displayResults(cands: Candidate[]): void {
      if (cands.length === 0) {
        statusEl.textContent = 'No candidates generated';
        return;
      }
      statusEl.textContent = `${cands.length} candidates found`;
      // Show top 3 results
      const top3 = cands.slice(0, 3);
      resultsEl.innerHTML = `
        <h3>Top Candidates</h3>
        <ol>
          ${top3.map((c, i) => `
            <li>
              <strong>#${i + 1}</strong> —
              Score: ${c.compositeScore.toFixed(2)},
              Height: ${c.estHeight.toFixed(2)}mm,
              Stability: ${c.stability},
              Overhang: ${c.overhangPenalty.toFixed(2)}
            </li>
          `).join('')}
        </ol>
        <p><em>Total: ${cands.length} candidates — ${cands.filter(c => c.stability === 'stable').length} stable</em></p>
      `;
      console.log('Candidates:', cands);
    }

    // --- Config controls ---
    const angleSlider = document.getElementById('angle-slider') as HTMLInputElement;
    const angleValue = document.getElementById('angle-value')!;
    angleSlider.addEventListener('input', () => {
      config.criticalAngleDeg = parseFloat(angleSlider.value);
      angleValue.textContent = angleSlider.value;
      // Re-run if we have data
      // (disabled for v1 — manual re-run via file reload)
    });

    // --- Boot ---
    boot();
    ```
