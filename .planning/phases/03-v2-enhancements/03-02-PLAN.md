       function applyQuaternionToVector(q: [number, number, number, number], v: [number, number, number]): [number, number, number] {
         // Standard v' = q * v * q^(-1) rotation
         const [w, x, y, z] = q;
         const [vx, vy, vz] = v;
         const uv_x = y * vz - z * vy;
         const uv_y = z * vx - x * vz;
         const uv_z = x * vy - y * vx;
         const uuv_x = y * uv_z - z * uv_y;
         const uuv_y = z * uv_x - x * uv_z;
         const uuv_z = x * uv_y - y * vx;
         return [
           vx + 2.0 * (w * uv_x + uuv_x),
           vy + 2.0 * (w * uv_y + uuv_y),
           vz + 2.0 * (w * uv_z + uuv_z),
         ];
       }
       function angularDistance(a: [number, number, number], b: [number, number, number]): number {
         const dot = a[0]*b[0] + a[1]*b[1] + a[2]*b[2];
         return Math.acos(Math.max(-1, Math.min(1, dot)));
       }
